495 [[nodiscard]] Matrix3D Adjoint() const {
496 // Calculate cofactors
497 float c00 = m11 * m22 - m12 * m21;
498 float c01 = m10 * m22 - m12 * m20;
499 float c02 = m10 * m21 - m11 * m20;
500
501 float c10 = m01 * m22 - m02 * m21;
502 float c11 = m00 * m22 - m02 * m20;
503 float c12 = m00 * m21 - m01 * m20;
504
505 float c20 = m01 * m12 - m02 * m11;
506 float c21 = m00 * m12 - m02 * m10;
507 float c22 = m00 * m11 - m01 * m10;
508
509 // Adjoint is the transpose of the cofactor matrix with appropriate sign adjustments
510 return Matrix3D(
511 c00, -c10, c20,
512 -c01, c11, -c21,
513 c02, -c12, c22
514 );
515 }

781 static Matrix3D RotationAxis(const Vector3D& axis, float angleRadians) {
782 float cosAngle = std::cos(angleRadians);
783 float sinAngle = std::sin(angleRadians);
784 float oneMinusCos = 1.0f - cosAngle;
785
786 // Ensure the axis is normalized
787 Vector3D normalizedAxis = axis.Normalize();
788 float x = normalizedAxis.x;
789 float y = normalizedAxis.y;
790 float z = normalizedAxis.z;
791
792 // Precompute common products
793 float xx = x * x;
794 float xy = x * y;
795 float xz = x * z;
796 float yy = y * y;
797 float yz = y * z;
798 float zz = z * z;
799
800 float xSin = x * sinAngle;
801 float ySin = y * sinAngle;
802 float zSin = z * sinAngle;
803
804 return Matrix3D(
805 cosAngle + xx * oneMinusCos, xy * oneMinusCos - zSin, xz * oneMinusCos + ySin,
806 xy * oneMinusCos + zSin, cosAngle + yy * oneMinusCos, yz * oneMinusCos - xSin,
807 xz * oneMinusCos - ySin, yz * oneMinusCos + xSin, cosAngle + zz * oneMinusCos
808 );
809 }

1038 [[nodiscard]] Vector3D CalculateEigenvalues() {
1039 // Ensure the matrix is symmetric
1040 if (!IsSymmetric()) {
1041 throw std::runtime_error("Eigenvalue calculation requires a symmetric matrix");
1042 }
1043
1044 // Use the characteristic equation for a 3x3 matrix
1045 // λ³ - tr(A)λ² + (m00m11 + m11m22 + m00m22 - m01² - m12² - m02²)λ - det(A) = 0
1046
1047 float a = -Trace();
1048 float b = m00 * m11 + m11 * m22 + m00 * m22 -
1049 m01 * m01 - m12 * m12 - m02 * m02;
1050 float c = -Determinant();
1051
1052 // This is a cubic equation. For simplicity, we'll use a numerical method
1053 // For a more robust implementation, consider using a dedicated eigenvalue solver
1054
1055 // For this example, we'll implement a simple power iteration method
1056 // which works well for symmetric matrices
1057
1058 Vector3D eigenvalues;
1059 Vector3D v = Vector3D(1, 1, 1).Normalize(); // Starting vector
1060
1061 for (int i = 0; i < 3; i++) {
1062 // Power iteration for largest eigenvalue
1063 for (int iter = 0; iter < 30; iter++) { // Usually converges in fewer iterations
1064 Vector3D new_v = (*this) * v;
1065 float length = new_v.Length();
1066
1067 if (length < 1e-10f) {
1068 // Matrix likely has a zero eigenvalue
1069 eigenvalues[i] = 0;
1070 break;
1071 }
1072
1073 new_v = new_v / length;
1074
1075 // Check for convergence
1076 if ((new_v - v).Length() < 1e-6f || (new_v + v).Length() < 1e-6f) {
1077 eigenvalues[i] = v.Dot((*this) * v); // Rayleigh quotient
1078 break;
1079 }
1080
1081 v = new_v;
1082
1083 if (iter == 29) {
1084 // Failed to converge
1085 throw std::runtime_error("Eigenvalue calculation failed to converge");
1086 }
1087 }
1088
1089 // Deflation - remove the found eigenvalue/eigenvector
1090 if (i < 2) { // We don't need to do this for the last eigenvalue
1091 Matrix3D vvT(
1092 v.x * v.x, v.x * v.y, v.x * v.z,
1093 v.y * v.x, v.y * v.y, v.y * v.z,
1094 v.z * v.x, v.z * v.y, v.z * v.z
1095 );
1096

1248 [[nodiscard]] Matrix3D Orthogonalize() const {
1249 // Extract column vectors
1250 Vector3D v1 = GetColumn(0);
1251 Vector3D v2 = GetColumn(1);
1252 Vector3D v3 = GetColumn(2);
1253
1254 // Gram-Schmidt process
1255 Vector3D u1 = v1.Normalize();
1256
1257 Vector3D u2 = v2 - u1 * v2.Dot(u1);
1258 float len2 = u2.Length();
1259 if (len2 > 1e-6f) {
1260 u2 = u2 / len2;
1261 } else {
1262 // Generate a vector orthogonal to u1
1263 if (std::fabs(u1.x) < std::fabs(u1.y)) {
1264 if (std::fabs(u1.x) < std::fabs(u1.z)) {
1265 u2 = Vector3D(1, 0, 0) - u1 * u1.x;
1266 } else {
1267 u2 = Vector3D(0, 0, 1) - u1 * u1.z;
1268 }
1269 } else {
1270 if (std::fabs(u1.y) < std::fabs(u1.z)) {
1271 u2 = Vector3D(0, 1, 0) - u1 * u1.y;
1272 } else {
1273 u2 = Vector3D(0, 0, 1) - u1 * u1.z;
1274 }
1275 }
1276 u2 = u2.Normalize();
1277 }
1278
1279 // u3 is orthogonal to both u1 and u2
1280 Vector3D u3 = u1.Cross(u2);
1281
1282 return Matrix3D(u1, u2, u3);
1283 }
